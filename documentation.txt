###############################  getMessages.php  ###############################

	требуемые данные от клиента
	user_uid		[string]	- пользовательский id. аунтификация в системе. 
	limit			[string]	- максимальное количество отдаваемых сообщений.
	before			[int]		- получить сообщения перед отправленным id.
    after			[int]		- получить сообщения после отправленного id.
	вы должны отправить либо before либо after!
	!!!	данные отдаем серверу через POST запрос.

	error 			[string]	- сообщаем клиенту статус "работы бекэнда". коды описаны в следующем блоке.
